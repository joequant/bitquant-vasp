\usepackage{tocloft}
 
\newcommand{\listcrossrefname}{List of Cross References}
\newlistof{crossref}{crf}{\listcrossrefname}
\newcommand{\crossref}[1]{%
  \refstepcounter{crossref}
  \par\noindent\textbf{Cross-reference \thecrossref. #1}
  \addcontentsline{crf}{crossref}
  {\protect\numberline{\thechapter.\thecrossref}#1}\par
}
